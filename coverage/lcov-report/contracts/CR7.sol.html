/// @dev Inherits standard ERC20 functionality from OpenZeppelin

    /// @notice Mints 1,000,000 CR7 tokens to the deployer on deployment
    constructor() ERC20("Cristiano Token", "CR7") {
        _mint(msg.sender, 1_000_000 ether);
    }
}
